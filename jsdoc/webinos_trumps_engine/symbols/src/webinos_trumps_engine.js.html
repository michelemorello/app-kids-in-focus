 98 function play()
 99 {
100    //I put container for status information in game.html - kwlodarska 14.03.2012 i.e. webinos_trumps.html Wei Guo 06-12-2012
101     gamet.started = true;
102 	document.getElementById("cont-button").innerHTML = '<a id="continue" class="button" onClick="cont();">Continue</a>';
103 	var newElem = document.getElementById("newElem");
104 	var newButton = document.getElementById("newButton");
105 	newElem.removeAttribute('class');
106 	newButton.setAttribute('onClick', 'newGame(); return false;');
107 
108 	var endElem = document.getElementById("endElem");
109 	endElem.removeAttribute('class');
110 
111 	rnd = 1 + (Math.floor(Math.random()*2));
112 	if(rnd === 1)
113 		{
114 		gamet.currentTurn = true;
115 		sendMsg('nextTurn', false);
116 		}
117 	else
118 		{
119 		gamet.currentTurn = false;
120 		sendMsg('nextTurn', true);
121 		}
122 	nextTurn();
123 };

125 /**
126  *	Compare players attribute values
127  *	@param {selection} Selection
128  */
129 
130 function evaluateAttr(selection)
131 {
132 	gamet.inputEnable = false;
133 	gamet.opponentCurrentCard = selection.card;
134 	showCard("2", gamet.opponentCurrentCard, 'flip');
135 	myValue = gamet.deck.data[gamet.myCurrentCard-1][selection.attr];
136 	opponentValue = gamet.deck.data[gamet.opponentCurrentCard-1][selection.attr];
137 
138 	//document.getElementById("status").innerHTML = "Attribute: "+selection.attr+" myValue: "+ myValue + " opponentValue: " + opponentValue;
139 
140 	if((myValue > opponentValue) || (myValue === opponentValue && !gamet.currentTurn))
141 	{
142 		//get boths cards
143 		gamet.currentTurn = true;
144 		gamet.myCards.push(gamet.myCurrentCard);
145 		gamet.myCards.push(gamet.opponentCurrentCard);
146  		document.getElementById("status-text").innerHTML = 'You win';
147 	}
148 	else
149 	{
150 		gamet.currentTurn = false;
151 		document.getElementById("status-text").innerHTML = 'You lose';
152 	}
153 
154 	setTimeout("nextTurn()",3000);
155 };

176 /**
177  *	Starts the next turn of the game
178  */
179 
180 function nextTurn()
181 {
182 	mark(null);
183 	gamet.inputEnable = true;
184 	gamet.sel = null;
185 	if(gamet.currentTurn)
186 		document.getElementById("cont-button").innerHTML = '<a id="continue" class="button" onClick="cont();">Continue</a>';
187 	else
188 		document.getElementById("cont-button").innerHTML = '';
189 
190 	if(gamet.numOfMyCards() === 0)
191 		{
192 		//I changes alert into status information - kwlodarska 14.03.2012
193 		document.getElementById("status-text").innerHTML = 'LOSER';
194       document.getElementById("status-text").className = 'loseColor';
195 		//document.getElementById("status").style.color = '#cb2929';
196 		setTimeout("anotherGame()",3000);
197 		}
198 	else if(gamet.numOfMyCards() === gamet.deck.numOfCards())
199 		{
200 		//I changes alert into status information - kwlodarska 14.03.2012
201 		document.getElementById("status-text").innerHTML = 'WINNER';
202       document.getElementById("status-text").className = 'winColor';
203 		setTimeout("anotherGame()",3000);
204 		}
205 	else
206 	{
207 		updateUI();
208 		gamet.myCards.sort(randOrd);
209 		gamet.myCurrentCard = gamet.myCards.shift();
210 		gamet.opponentCurrentCard = null;
211 		showCard("1",gamet.myCurrentCard);
212 		switchUICard("2", "b", "");
213 		//activateAttributes of the player who's the turn
214 	}
215 
216 };

247 /**

256 function switchUICard(player, show, hide, flip)
257 {
258 	if(player == 2) {
259 		if(flip) {
260 			document.getElementById("cardcontainer2").className += " flip-animate";
261 		} else {
262 			var flipClassArr = document.getElementById("cardcontainer2").className.split(" ");
263 			var flipClassIndex = flipClassArr.indexOf("flip-animate");
264 			if(flipClassIndex != -1) {
265 				flipClassArr.splice(flipClassIndex,1);
266 				document.getElementById("cardcontainer2").className = flipClassArr.join(" ");
267 			}
268 		}
269 	}
270 	document.getElementById("card"+player+show).style.visibility = "visible";
271 	document.getElementById("card"+player+hide).style.visibility = "hidden";
272 };

274 /**
275  *	Initializes players cards
276  *	@param {deckCards} deckCards
277  *	@return {int[]} cardsIds
278  */
279 
280 function myCards(deck)
281 {
282 	var myCards = new Array();
283 	if(useMyDeck)
284 	{
285 		var opponentCards = new Array();
286 		for(i=1; i<=deck.data.length; i++)
287 			opponentCards.push(i);
288 
289 		//If the number of cards is odd, one is randomly dropped
290 		if(opponentCards.length%2 !== 0)
291 			{
292 				opponentCards.sort(randOrd);
293 				drop = opponentCards.shift();
294 				console.log("DROPPED: " + drop);
295 			}
296 		var ind = null;
297 		do {
298 			ind = Math.floor(Math.random()*opponentCards.length);
299 			myCards.push(opponentCards[ind]);
300 			opponentCards.splice(ind, 1);
301 		}while(myCards.length!=Math.floor(deck.data.length/2));
302 		/*console.log("my Cards: "+myCards);
303 		console.log("his Cards: "+opponentCards);*/
304 
305 		//send opponentCards
306 		sendMsg('myCards', opponentCards);
307 	}
308 	return myCards;
309 };

327 /**
328  *	@class Represents the object Deck
329 */
330 
331 var deckCards = function(){
332 	/**
333 	 *	Deck cards information
334 	 */
335 	this.data = cardData;
336 	this.attributes = initializeAttributes(this.data);
337 	/**
338 	 *	Returns the number of the card in the deck
339 	 *	@return {int} numberOfCards
340 	 */
341 	this.numOfCards = function(){
342 		if(gamet.deck.data.length % 2 === 0)
343 			return gamet.deck.data.length;
344 		else
345 			return gamet.deck.data.length-1;
346 	}
347 
348 	/**
349 	 *	Return the number of attributes of a deck card
350 	 *	@return {int} numberOfAttributes
351 	 */
352 	this.getNumAtts = function(){
353     		return this.attributes.length;
354   	}
355 	this.backCard = "";
356 };
